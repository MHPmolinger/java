var与let区别
  1. 声明后未赋值，表现相同
    2. var varTest; let letTest; console.log(varTest); //undefined console.log(letTest); //undefined
  2. 使用未声明的变量，表现不同
    3. console.log(varTest); //undefined console.log(letTest); //直接报错
  3. 重复声明同一个变量时，表现不同
    4. var varTest = ' OK'; let letTest = 'OK'; var varTest = 'changed'; let letTest = 'changed'; //直接报错
  4. 变量作用范围，表现不同
    5. var varTest = 'OK'; let letTest = 'OK'; { var varTest = 'changed'; let letTest = 'changed'; } console.log(varTest); //"changed" console.log(letTest); //"OK"
